SensorServo board



Weekly assignment: add an output device to a microcontroller board you've designed and program it to do something.

I decided to experiment with the MQ7 gas sensor, because ti will be necessary for my final project. Thje MQ7 will be useful as input for my board, to which I will connect some actuators as servo motor and LED.



First, I downloaded the MQ7 datasheet . I discovered that you can find this sensor in two versions: one shows VCC, GND and just one digital Pin. It basically used for alarm systems. After a certain level of Carbon Dioxyde in the air, the sensor provides the alarm. 






    
SensorServo.ino 

#include 
Servo myServo;
//int servo = 9; 
int MQ7 = A1; 
int gas = 0;
int valgas = 0;
int ambient = 0;
void setup () {
  Serial.begin (9600);
  myServo.attach (9);
  // pinMode (servo, OUTPUT);
  //analogWrite (servo, 0);
}
void loop () {
  // MQ7 reads data and myServo behaves accordingly
  valgas = analogRead(MQ7);
  Serial.println(valgas);
  if (valgas <= 200)
    {
      myServo.write (90);
      delay (10000); ;
      delay (10000);
    }
   else if (valgas > 200 && valgas < 250) 
    {                                                  
      myServo.write (45); 
      delay (10000); 
      myServo.write (0);
      delay (10000);
    }
   else if (valgas > 250)
    {
      myServo.write (120); 
      delay (10000); 
      myServo.write (0);
      delay (10000);
    }
}

float MQ7 = A1; 
int gas = 0; 
int valgas = 0; 
int ambient = 0;
int LED = 9;
void setup () {
  Serial.begin (9600);
  pinMode (LED, OUTPUT);
  //analogWrite (servo, 0); 
}
void loop () {
  // We read the gas data, and behave accordingly
  valgas = analogRead(MQ7);
  Serial.println(valgas);
  if (valgas <= 200)
    {
      digitalWrite (LED, LOW); 
      delay (1000); 
    }
   else if (valgas > 200) 
   {
     digitalWrite (LED, HIGH); 
      delay (10000); 
   }
   
}
